        function initProjectSubnav() {
            const projectNavMarkup = `
                <nav class="project-subnav" aria-label="Project page navigation">
                    <a href="#project-overview">Overview</a>
                    <a href="#project-analysis">Analysis</a>
                    <a href="#project-design">System Design</a>
                    <a href="#project-implementation">Implementation</a>
                    <a href="#project-testing">Testing</a>
                </nav>
            `;

            projectSubpages.forEach(pageId => {
                const page = document.getElementById('page-' + pageId);
                if (page && !page.querySelector('.project-subnav')) {
                    page.insertAdjacentHTML('afterbegin', projectNavMarkup);
                }
            });
        }

        function setupHashLinks() {
            document.querySelectorAll('a[href^="#"]').forEach(link => {
                link.addEventListener('click', event => {
                    const pageId = link.getAttribute('href').substring(1);
                    const target = document.getElementById('page-' + pageId);

                    if (target) {
                        event.preventDefault();
                        window.location.hash = pageId;
                        navigateTo(pageId);
                    }
                });
            });
        }

        function navigateTo(pageId) {
            // Close mobile menu first if open
            document.getElementById('nav-menu').classList.remove('mobile-open');
            
            // Hide all views
            const pages = document.querySelectorAll('.portfolio-page');
            pages.forEach(page => {
                page.classList.remove('active');
            });

            // Show active target
            const activePage = document.getElementById('page-' + pageId);
            if (activePage) {
                activePage.classList.add('active');
            }

            // Update navbar items
            const navLinks = document.querySelectorAll('.nav-link, .dropdown-item');
            navLinks.forEach(link => {
                link.classList.remove('active');
            });

            // Mark active element
            const activeLink = document.getElementById('link-' + pageId);
            if (activeLink) {
                activeLink.classList.add('active');
            }

            // Handle Project Dropdown trigger highlighting
            const projectTrigger = document.getElementById('project-dropdown-trigger');
            if (projectSubpages.includes(pageId)) {
                projectTrigger.classList.add('active');
            } else {
                projectTrigger.classList.remove('active');
            }

            // Scroll cleanly to top
            window.scrollTo({ top: 0, behavior: 'smooth' });

            // Trigger Count-Up stats if switching to home page
            if (pageId === 'home') {
                setTimeout(initCountUp, 300);
            }

            updateProjectSubnav(pageId);
        }

        // Router listener routing dynamic window hashes
        function handleRouting() {
            const hash = window.location.hash.substring(1);
            const validPages = ['home', 'about', 'learning', 'project-overview', 'project-analysis', 'project-design', 'project-implementation', 'project-testing', 'certification', 'reflections'];
            
            if (validPages.includes(hash)) {
                navigateTo(hash);
            } else {
                navigateTo('home');
                window.location.hash = 'home';
            }
        }

        window.addEventListener('hashchange', handleRouting);
        window.addEventListener('DOMContentLoaded', () => {
            initProjectSubnav();
            setupHashLinks();
            handleRouting();
        });

        /* =========================================
           4. VISUAL STATISTICS COUNT-UP EFFECT
           ========================================= */
        let countUpExecuted = false;
        function initCountUp() {
            if (countUpExecuted) return;
            const stats = document.querySelectorAll('.stat-number-label[data-target]');
            
            stats.forEach(stat => {
                const target = parseInt(stat.getAttribute('data-target'), 10);
                const isPlus = stat.innerText.includes('+');
                let count = 0;
                const increment = Math.ceil(target / 50);
                const timer = setInterval(() => {
                    count += increment;
                    if (count >= target) {
                        count = target;
                        clearInterval(timer);
                    }
                    stat.innerText = count + (isPlus ? '+' : '');
                }, 30);
            });
            countUpExecuted = true;
        }

           5. INTERACTIVE SCHEMATIC CANVAS BACKGROUND
           ========================================= */
        const canvas = document.getElementById('circuit-canvas');
        const ctx = canvas.getContext('2d');

        let width = canvas.width = window.innerWidth;
        let height = canvas.height = window.innerHeight;

        window.addEventListener('resize', () => {
            width = canvas.width = window.innerWidth;
            height = canvas.height = window.innerHeight;
        });

        class Node {
            constructor(x, y) {
                this.x = x;
                this.y = y;
                this.baseX = x;
                this.baseY = y;
                this.radius = Math.random() * 2 + 1.5;
                this.pulse = Math.random() * Math.PI;
                this.speed = 0.02;
            }

            draw() {
                this.pulse += this.speed;
                const alpha = 0.15 + Math.sin(this.pulse) * 0.08;
                
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.radius + Math.sin(this.pulse) * 0.8, 0, Math.PI * 2);
                ctx.fillStyle = `rgba(0, 212, 255, ${alpha})`;
                ctx.shadowBlur = 10;
                ctx.shadowColor = 'rgba(0, 212, 255, 0.4)';
                ctx.fill();
                ctx.shadowBlur = 0; // Reset blur
            }

            update() {
                // Subtle organic floating movement
                this.x = this.baseX + Math.sin(this.pulse) * 4;
                this.y = this.baseY + Math.cos(this.pulse) * 4;
            }
        }

        const nodesList = [];
        const maxNodes = 65;

        // Generate grid-like nodes mirroring circuit points
        for (let i = 0; i < maxNodes; i++) {
            nodesList.push(new Node(
                Math.random() * width,
                Math.random() * height
            ));
        }

        function drawConnections() {
            ctx.strokeStyle = 'rgba(0, 212, 255, 0.035)';
            ctx.lineWidth = 1;
            
            for (let i = 0; i < nodesList.length; i++) {
                for (let j = i + 1; j < nodesList.length; j++) {
                    const dist = Math.hypot(nodesList[i].x - nodesList[j].x, nodesList[i].y - nodesList[j].y);
                    
                    if (dist < 180) {
                        ctx.beginPath();
                        ctx.moveTo(nodesList[i].x, nodesList[i].y);
                        
                        // Trace orthogonal line segments resembling PCB wires
                        const midX = (nodesList[i].x + nodesList[j].x) / 2;
                        ctx.lineTo(midX, nodesList[i].y);
                        ctx.lineTo(midX, nodesList[j].y);
                        ctx.lineTo(nodesList[j].x, nodesList[j].y);
                        ctx.stroke();
                    }
                }
            }
        }

        function animateBgCanvas() {
            ctx.clearRect(0, 0, width, height);
            
            drawConnections();
            
            nodesList.forEach(node => {
                node.update();
                node.draw();
            });
            
            requestAnimationFrame(animateBgCanvas);
        }

        animateBgCanvas();
